197 bool Cost_Bigger_Node(int custNum){
198  //贪心计算可能的最小流量花费
199  EdgeNode *p;
200  int sumBand = 0, sumCost = 0;
201  int minRent = MY_INT_MAX;
202  int band = 0;
203  int caldown = 0;
204  while(sumBand < graph.customerInfo[custNum].bandNeed){
205  p = graph.adjList[graph.customerInfo[custNum].toNodeID].outEdgeTails;
206  while(p && (p -> tailID < graph.nodeNum)){
207  //if(p -> vexOut -> vexOut)
208  if(p ->Rental < minRent && p ->rest_bandWidth != 0){
209  band = p ->bandWidth;
210  minRent = p ->Rental;
211  }
212  p = p ->vexOut;
213  }
214  p = graph.adjList[graph.customerInfo[custNum].toNodeID].outEdgeTails;
215  while(p && (p -> tailID < graph.nodeNum)){
216  if(p ->Rental == minRent && p ->bandWidth == band && p ->rest_bandWidth != 0){
217  if(sumBand + band < graph.customerInfo[custNum].bandNeed){
218  p ->rest_bandWidth = 0;
219  sumBand += band;
220  sumCost += minRent * band;
221  minRent = MY_INT_MAX;
222  break;
223  }
224  else{
225  sumCost += (graph.customerInfo[custNum].bandNeed - sumBand) * p ->Rental;
226  caldown = 1;
227  break;
228  }
229 
230  }
231  p = p ->vexOut;
232  }
233  if(caldown)
234  break;
235  }
236  //恢复边信息
237  p = graph.adjList[graph.customerInfo[custNum].toNodeID].outEdgeTails;
238  while(p){
239  p ->rest_bandWidth = p ->bandWidth;
240  p = p ->vexOut;
241  }
242  return (sumCost >= graph.serverCost) ? true : false;
243 }
