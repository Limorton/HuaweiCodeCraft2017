130 bool SPFA(int startNode, int endNode){
131  for(int i = 0; i < graph.nodeNum + 2; ++i){
132  distances[i] = DIST_MAX;
133  inQueue[i] = false;
134  preNode[i] = -1;
135  }
136  distances[startNode] = 0;
137  inQueue[startNode] = true;
138  Releax.push_front(graph.adjList[startNode]);
139  VertexNode u;
140  while(!Releax.empty()){
141  u = Releax.front();
142  Releax.pop_front();
143  inQueue[u.nodeID] = false;
144  EdgeNode *p = graph.adjList[u.nodeID].outEdgeTails;
145  while(p){
146  if(p ->rest_bandWidth > 0 && distances[p ->tailID] > distances[p ->headID] + p ->Rental){
147  distances[p ->tailID] = distances[p ->headID] + p ->Rental;
148  preNode[p ->tailID] = p ->headID;
149 
150  if(!inQueue[p ->tailID]){
151  inQueue[p ->tailID] = true;
152  if(!Releax.empty()){ //队列优化SLF策略，该点距离小于队首点的距离则插入队首，否则插入队尾
153  if(distances[p ->tailID] < distances[Releax.front().nodeID])
154  Releax.push_front(graph.adjList[p ->tailID]);
155  else
156  Releax.push_back(graph.adjList[p ->tailID]);
157  }else{
158  Releax.push_back(graph.adjList[p ->tailID]);
159  //cout << "push in: " << p ->tailID << endl;
160  }
161  }
162  }
163  p = p -> vexOut;
164  }
165  }
166  if(distances[endNode] == DIST_MAX) //说明不存在到达endnode 的路径
167  return false;
168  return true;
169 }

265 int MinCostMaxFlow(int startNode, int endNode){
266  Init_Variables();
267  int cost = 0;
268  int flows = 0;
269  while(SPFA(startNode, endNode)){
270  oneFlow = FLOW_MAX;
271  //找到该次SPFA搜索到的最短路径上的最小带宽
272  for(int v = endNode; v != startNode; v = preNode[v]){
273  int u = preNode[v];
274  EdgeNode *p = graph.adjList[u].outEdgeTails;
275  while(p ->tailID != v){
276  p = p ->vexOut;
277  }
278  if(p ->rest_bandWidth < oneFlow){
279  oneFlow = p ->rest_bandWidth;
280  }
281  }
282  flows += oneFlow;
283  cost += distances[endNode] * oneFlow;
284  //更新该路径上的残余带宽
285  for(int v = endNode; v != startNode; v = preNode[v]){
286  int u = preNode[v];
287  EdgeNode *p = graph.adjList[u].outEdgeTails;
288  //正向边容量减少
289  while(p ->tailID != v){
290  p = p ->vexOut;
291  }
292  p ->rest_bandWidth -= oneFlow;
293  //反向边容量增加
294  EdgeNode *q = graph.adjList[p ->tailID].outEdgeTails;
295  while(q ->tailID != u){
296  q = q ->vexOut;
297  }
298  q ->rest_bandWidth += oneFlow;
299  }
300  Update_Route(startNode, endNode);
301  }
302  return cost;
303 }
